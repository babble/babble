50:       line.gsub!(/(^\s*)context_teardown(\s*[do|\{])/, '\1after(:all)\2')

55:         pre = $1
56:         should = $2
57:         post = $3
58:         be_or_equal = post =~ /(<|>)/ ? "be" : "equal"
59:

61:       end
